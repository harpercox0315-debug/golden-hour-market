// --- Config --- const API_BASE = "http://localhost:5000/api"; // Replace with your deployed backend URL let currentUser = null; // --- User Signup / Login --- async function signup(username){ const res = await fetch(`${API_BASE}/users`, { method: "POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({username}) }); const data = await res.json(); currentUser = { username, uid: data.uid }; alert(`Logged in as ${username}`); } // --- Load Marketplace --- async function loadItems(){ const res = await fetch(`${API_BASE}/items`); const items = await res.json(); const container = document.getElementById("browseList"); container.innerHTML = ""; items.forEach(item=>{ const card = document.createElement("div"); card.className = "card"; card.innerHTML = `

✨ ${item.name}

${item.desc}

Price: $${item.price}

Status: ${item.status}

Seller: ${item.sellerID}

`; // Add images const imgDiv = card.querySelector(`#images-${item._id}`); item.images.forEach(src=>{ const img = document.createElement("img"); img.src = src; img.style.width="80px"; img.style.marginRight="5px"; imgDiv.appendChild(img); }); container.appendChild(card); }); } // --- Post Item --- async function postItem(name, desc, price, images){ if(!currentUser) return alert("Sign in first!"); const res = await fetch(`${API_BASE}/items`, { method: "POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({ name, desc, price, images, sellerID:currentUser.uid }) }); const data = await res.json(); alert("Item posted!"); loadItems(); } // --- Buy Item --- async function buyItem(itemID){ if(!currentUser) return alert("Sign in first!"); const res = await fetch(`${API_BASE}/items/${itemID}/buy`, { method:"POST" }); const data = await res.json(); alert(data.message); loadItems(); } // --- Chat --- async function openChat(itemID){ const chatDiv = document.getElementById("chatDiv"); chatDiv.innerHTML = `

Chat for item ${itemID}

`; const input = document.createElement("input"); input.placeholder = "Type message..."; const sendBtn = document.createElement("button"); sendBtn.innerText = "Send"; sendBtn.onclick = async ()=>{ const text = input.value; if(!text) return; await fetch(`${API_BASE}/items/${itemID}/chat`, { method:"POST", headers:{"Content-Type":"application/json"}, body: JSON.stringify({ userID: currentUser.uid, text }) }); input.value=""; loadChat(itemID); }; chatDiv.appendChild(input); chatDiv.appendChild(sendBtn); loadChat(itemID); } async function loadChat(itemID){ const res = await fetch(`${API_BASE}/items/${itemID}/chat`); const messages = await res.json(); const chatDiv = document.getElementById("chatDiv"); const msgList = document.createElement("div"); msgList.innerHTML = ""; messages.forEach(msg=>{ const p = document.createElement("p"); p.innerText = `${msg.userID}: ${msg.text}`; msgList.appendChild(p); }); chatDiv.appendChild(msgList); } // --- Wishlist --- async function addWishlist(itemID){ if(!currentUser) return alert("Sign in first!"); const res = await fetch(`${API_BASE}/users/${currentUser.uid}/wishlist`, { method:"POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({ itemID }) }); const data = await res.json(); alert("Added to wishlist!"); } // --- Meetup --- async function setMeetup(itemID, lat, lng){ const res = await fetch(`${API_BASE}/items/${itemID}/meetup`, { method:"POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({ lat, lng }) }); const data = await res.json(); alert("Meetup set!"); } // --- Accept / Decline Trade --- async function respondTrade(itemID, action){ const res = await fetch(`${API_BASE}/items/${itemID}/trade`, { method:"POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({ action }) }); const data = await res.json(); alert(data.message); loadItems(); } // --- Initialize --- document.addEventListener("DOMContentLoaded", ()=>{ const username = prompt("Enter your username to sign in:"); signup(username); loadItems(); });
